<h1>Сериализация и десериализация объектов</h1>
<h3>Создание сериализуемого класса</h3>
<pre><code class="language-java">import java.io.Serializable;

public class Person implements Serializable {
    private static final long serialVersionUID = 1L;
    private String name;
    private int age;

    public Person(String name, int age) {
        this.name = name;
        this.age = age;
    }

    @Override
    public String toString() {
        return &quot;Person{name='&quot; + name + &quot;', age=&quot; + age + &quot;}&quot;;
    }
}
</code></pre>
<h3>Сериализация объекта</h3>
<pre><code class="language-java">import java.io.FileOutputStream;
import java.io.IOException;
import java.io.ObjectOutputStream;

public class SerializationExample {
    public static void main(String[] args) {
        Person person = new Person(&quot;Alice&quot;, 30);

        try (FileOutputStream fos = new FileOutputStream(&quot;person.ser&quot;);
             ObjectOutputStream oos = new ObjectOutputStream(fos)) {
            oos.writeObject(person);
        } catch (IOException e) {
            System.out.println(&quot;Произошла ошибка: &quot; + e.getMessage());
        }
    }
}
</code></pre>
<h3>Десериализация объекта</h3>
<pre><code class="language-java">import java.io.FileInputStream;
import java.io.IOException;
import java.io.ObjectInputStream;

public class DeserializationExample {
    public static void main(String[] args) {
        try (FileInputStream fis = new FileInputStream(&quot;person.ser&quot;);
             ObjectInputStream ois = new ObjectInputStream(fis)) {
            Person person = (Person) ois.readObject();
            System.out.println(person);
        } catch (IOException | ClassNotFoundException e) {
            System.out.println(&quot;Произошла ошибка: &quot; + e.getMessage());
        }
    }
}
</code></pre>
