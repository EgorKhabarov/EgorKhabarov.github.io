class WrapperMap:
    def __init__(self, d: dict):
        self.d = d

    def __getattr__(self, item: str):
        value = self.d.get(item)
        if isinstance(value, dict):
            return self.__class__(value)
        return value

    def __setattr__(self, name, value):
        if name == "d":
            super().__setattr__(name, value)
        else:
            self.d[name] = value

    def __repr__(self):
        return repr(self.d)
